    <script>

        // Mobile menu toggle

        const mobileMenuButton = document.getElementById('mobile-menu-button');

        const mobileMenu = document.getElementById('mobile-menu');

        mobileMenuButton.addEventListener('click', () => {

        // Interactive Plan Builder Logic

        const planForm = document.getElementById('planForm');

        const kpiContainer = document.getElementById('kpi-options');

        const downloadBtn = document.getElementById('downloadPlan');


        const kpis = {

            Awareness: ['Reach', 'Frequency', 'Ad Recall', 'Brand Awareness'],

            Consideration: ['View-Through Rate (VTR)', 'Clicks', 'Engagement Rate', 'Consideration Lift'],

            Action: ['Conversions', 'Cost Per Acquisition (CPA)', 'Return On Ad Spend (ROAS)', 'Sales Lift']

        };


        function updateKpiOptions() {

            const selectedObjective = document.querySelector('input[name="objective"]:checked').value;

            kpiContainer.innerHTML = '';

            kpis[selectedObjective].forEach(kpi => {

                const label = document.createElement('label');

                label.className = 'inline-flex items-center';

                label.innerHTML = `

                    <input type="checkbox" class="form-checkbox h-4 w-4 text-blue-600 border-gray-400 rounded focus:ring-blue-500" name="kpi" value="${kpi}">

                    <span class="ml-3 text-gray-700">${kpi}</span>

                `;

                kpiContainer.appendChild(label);

            });

        }


        planForm.addEventListener('change', updateKpiOptions);


        downloadBtn.addEventListener('click', () => {

            const campaignName = document.getElementById('campaignName').value || 'Untitled Campaign';

            const objective = document.querySelector('input[name="objective"]:checked').value;

            const selectedKpis = Array.from(document.querySelectorAll('input[name="kpi"]:checked')).map(el => el.value);


            let planSummary = `

MEDIA PLAN SUMMARY

==================


Campaign Name: ${campaignName}

Primary Objective: ${objective}


Recommended YouTube Modules:

`;

            

            if (objective === 'Awareness') {

                planSummary += "- Awareness Module (VRC + Non-skip)\n";

            } else if (objective === 'Consideration') {

                planSummary += "- Awareness Module (VRC + Non-skip)\n- Engagement Module (VVC + Partnership Ads)\n";

            } else { // Action

                planSummary += "- Awareness Module (VRC + Non-skip)\n- Engagement Module (VVC + Partnership Ads)\n- Online Sales Module (Demand Gen)\n";

            }


            planSummary += `

Key KPIs to Track:

${selectedKpis.length > 0 ? selectedKpis.map(kpi => `- ${kpi}`).join('\n') : '- None selected'}


==================

This summary was generated by the Google & YouTube Media Plan Builder.

            `;


            const blob = new Blob([planSummary.trim()], { type: 'text/plain' });

            const link = document.createElement('a');

            link.href = URL.createObjectURL(blob);

            link.download = `${campaignName.replace(/\s+/g, '_')}_Plan_Summary.txt`;

            document.body.appendChild(link);

            link.click();

            document.body.removeChild(link);

        });


        // Initial call to populate KPIs

        updateKpiOptions();

    </script>
